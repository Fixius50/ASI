Fase 2: Instalación Automatizada

Despliegue mediante Script Bash

Paso 1: Crear Script
Script Oficial
setup_full.sh
#!/bin/bash set -e # --- VARIABLES OFICIALES --- MODEL_ID="gpt-oss:20b" OPENCODE_URL="https://github.com/anomalyco/opencode/releases/latest/download/OpenCode-Darwin-arm64.dmg" echo "========== INICIANDO INSTALACIÓN (IP: 192.168.7.174) ==========" # 1. INSTALACIÓN DE OLLAMA (BACKEND) echo " [1/4] Instalando Ollama..." if ! command -v ollama &>/dev/null; then curl -fsSL https://ollama.com/install.sh | sh fi # 2. CONFIGURACIÓN DE RED (CRÍTICO) echo " [2/4] Abriendo puertos de red (0.0.0.0)..." launchctl setenv OLLAMA_HOST "0.0.0.0" # 3. INSTALACIÓN DE OPENCODE (FRONTEND) echo " [3/4] Instalando OpenCode..." curl -L -o /tmp/OpenCode.dmg "$OPENCODE_URL" hdiutil attach /tmp/OpenCode.dmg -nobrowse -mountpoint /Volumes/OpenCodeInstaller -quiet sudo cp -R /Volumes/OpenCodeInstaller/OpenCode.app /Applications/ sudo xattr -cr /Applications/OpenCode.app # Bypass Gatekeeper hdiutil detach /Volumes/OpenCodeInstaller -quiet rm /tmp/OpenCode.dmg # 4. DESCARGA DEL MODELO echo " [4/4] Descargando modelo $MODEL_ID..." open -a Ollama sleep 5 ollama pull $MODEL_ID echo "☑ INSTALACIÓN COMPLETADA."
